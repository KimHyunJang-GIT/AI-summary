# InfoPilot 프로젝트의 Python 의존성 목록
# 각 섹션은 라이브러리의 주요 사용처를 나타냅니다.

# ===================================
# 1. UI (사용자 인터페이스) 관련
# ===================================
customtkinter # 커스텀 가능한 Tkinter 기반 GUI 프레임워크
requests        # FastAPI 백엔드와 통신하기 위한 HTTP 요청 라이브러리

# ===================================
# 2. Core (핵심 로직 및 데이터 처리) 관련
# ===================================
pandas              # 데이터 처리 및 분석 (DataFrame)
joblib              # Python 객체 직렬화/역직렬화 (모델 저장/로드)
tqdm                # 진행률 바 표시 (콘솔 및 UI 스레드)
python-docx         # .docx 파일 텍스트 추출
python-pptx         # .pptx 파일 텍스트 추출
pyhwp               # .hwp 파일 텍스트 추출
pywin32             # Windows 시스템 관련 기능 (예: 드라이브 목록 조회)에 필요
pypandoc            # 다양한 문서 형식 변환 (예: .doc, .ppt 파일 텍스트 추출에 사용)
pymupdf             # PDF 파일 텍스트 및 이미지 추출 (OCR 연동)
sentence-transformers # 의미론적 임베딩 생성 (문서 및 질의)
Pillow              # 이미지 처리 (EasyOCR의 의존성)
openpyxl            # .xlsx, .xlsm 파일 텍스트 추출
xlrd                # .xls 파일 텍스트 추출
EasyOCR             # 이미지 내 텍스트 인식 (OCR)
pyarrow             # pandas의 parquet 파일 처리에 필요
rank_bm25           # BM25 알고리즘 기반 키워드 검색
torch               # PyTorch 딥러닝 프레임워크 (sentence-transformers, EasyOCR의 의존성)

# ===================================
# 3. FastAPI (백엔드 API) 관련
# ===================================
uvicorn             # ASGI 서버 구현 (FastAPI 애플리케이션 실행)
fastapi             # 웹 API 구축 프레임워크

# ===================================
# 4. LLM (대규모 언어 모델) 관련
# ===================================
google-generativeai # Google Gemini API 연동
transformers        # Hugging Face 로컬 LLM 연동
sentencepiece       # Hugging Face 모델 (특히 KoGPT2)의 토크나이저 의존성
